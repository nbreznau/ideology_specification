std_coefficients <- summary(m_4_beta)$coefficients[, "Standardized"]
# Convert the standardized coefficients into a data frame for easy joining
coef_standardized <- data.frame(
  term = names(std_coefficients),  # Names of the coefficients (e.g., group1, group3)
  std_estimate = std_coefficients  # The standardized coefficients
)

# Filter for the specific terms (group1, group3, pbelief, stats_brw, topic_brw)
coef_filtered <- coef_df %>%
  filter(term %in% c("topic_brw", "stats_brw", "pbelief", "group3", "group1")) %>%
  left_join(coef_standardized, by = "term") %>%  # Join with standardized coefficients
  mutate(term = factor(term, levels = c("group1", "group3", "pbelief", "stats_brw", "topic_brw")))

Predicted values at means

# Step 1: Calculate means of the variables to be fixed
mean_stats_brw <- mean(df$stats_brw, na.rm = TRUE)
mean_topic_brw <- mean(df$topic_brw, na.rm = TRUE)
mean_t2 <- mean(df$t2, na.rm = TRUE)
mean_t3 <- mean(df$t3, na.rm = TRUE)
mean_mdegree <- mean(as.numeric(df$mdegree), na.rm = TRUE)
mean_pbelief <- mean(as.numeric(df$pbelief), na.rm = TRUE)

# Step 2: Create a new dataset for prediction
# Keep group1 and group3 at their actual values, and others at their means
new_data <- data.frame(
  group1 = df$group1,
  group3 = df$group3,
  stats_brw = mean_stats_brw,
  topic_brw = mean_topic_brw,
  t2 = mean_t2,
  t3 = mean_t3,
  pbelief = mean_pbelief,
  index = df$index,
  mdegree = 1)

# Step 3: Predict using the new data
df$AME_hat <- predict(m_4, newdata = new_data)

#mean center, as it has no natural metric
df$AME_hat = df$AME_hat - mean(df$AME_hat)
